```bash
npm install --save react-button-group-tis
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-button-group-tis'
import 'react-button-group-tis/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
